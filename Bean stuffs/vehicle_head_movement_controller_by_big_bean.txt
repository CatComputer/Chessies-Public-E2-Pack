#Functions:
#----------
#Returns the difference between a vector passed to this call from the vector passed in the previous call. The difference is returned per-coordinate.
function vector tickDiff(ID:string, Var:vector) {
    local SID = "vt_"+ID
    if (!FUNC_DATA:exists(SID)) {
        FUNC_DATA[SID, vector] = Var
    }

    local Delta = (Var - FUNC_DATA[SID, vector])
    FUNC_DATA[SID, vector] = Var
    return Delta
}
function number tickDiff(ID:string, Var:number){
    ID = "nt_" + ID
    if(!FUNC_DATA:exists(ID)){
        FUNC_DATA[ID, number] = Var
    }
    local Delta = (Var - FUNC_DATA[ID, number])
    FUNC_DATA[ID, number] = Var
    return Delta
}

#Smooths a vector over time. The smoothing is performed per-coordinate, e.g. the x coordinate is smoothed independently of the y coordinate.
function vector vecSmooth(Val:vector, SmoothLevel:number, ID:string, Init:vector) {
    local SID = "vs_"+ID
    if (!FUNC_DATA:exists(SID)) {
        FUNC_DATA[SID, vector] = Init
    }

    local LastVal = FUNC_DATA[SID, vector]
    LastVal += (Val - LastVal) / max(SmoothLevel, 1)
    FUNC_DATA[SID, vector] = LastVal
    return LastVal
}

#Adds a deadzone to a vector, i.e. a volume within which a change to the input vector causes no change to the output vector.
function vector vecDeadzone(Val:vector, FrontBoundary:vector, RearBoundary:vector){
    return maxVec(Val - FrontBoundary, vec()) + minVec(Val - RearBoundary, vec())
}

#Used to smoothly transition between user-defined camera positions.
function void interpolateCameraMovement(CamID:number, SetPos:vector){
    local SID = "cam_lerp_"+CamID
    if(!FUNC_DATA:exists(SID)){
        FUNC_DATA[SID, array] = array(SetPos, SetPos, 0)
    }
    local LOCAL_DATA = FUNC_DATA[SID, array]
    local Position = LOCAL_DATA[1, vector]
    local MoveSpeed = LOCAL_DATA[3, number]
    #Every time the user changes the desired position, reset the movement so that it smoothly interpolates from the current position
    if(LOCAL_DATA[2, vector] != SetPos){
        LOCAL_DATA[2, vector] = SetPos
        MoveSpeed = 0
    }

    MoveSpeed += min(1 - MoveSpeed, Tickrate)
    Position += (SetPos - Position) * MoveSpeed / SeatPositionSmoothing

    LOCAL_DATA[1, vector] = Position
    LOCAL_DATA[3, number] = MoveSpeed
}



#Connects the respective outputs from this E2 to the inputs of the passed camera controller.
function number wirelink:setUpCamCtrlWires(ID:number) {
    if (!This | !This:entity():isValid()) {
        return 0
    }

    This:entity():createWire(entity(), "Activated", format("C%d_Active", ID))
    This:entity():createWire(entity(), "Position", format("C%d_Pos", ID))
    This:entity():createWire(entity(), "Parent", format("C%d_PARENT", ID))

    return 1
}

#Creates a camera parent hologram in its default position, relative to the entity passed to this function. This entity should be the seat that the camera is related to.
function void entity:createCameraParent(ID:number) {
    if (!This) {
        return
    }

    holoCreate(ID, This:toWorld(vec(0, 0, 48)), vec(1), This:toWorld(ang(0, 0, 0)), vec(255), (Debug ? "models/sprops/misc/axis_plane.mdl" : "models/sprops/misc/empty.mdl"))
    holoParent(ID, This)

    switch(ID){
    default,
    case 1,
        C1_PARENT = holoEntity(ID)
        C1_Active = 0
        C1_Pos = C1_PARENT:pos()
        break
    case 2,
        C2_PARENT = holoEntity(ID)
        C2_Active = 0
        C2_Pos = C2_PARENT:pos()
        break
    case 3,
        C3_PARENT = holoEntity(ID)
        C3_Active = 0
        C3_Pos = C3_PARENT:pos()
        break
    }
}

#Does the funny stuff to the camera specified by ID. You should pass the seat entity that's coupled to the given camera to this function, otherwise the behavior is undefined.
function vector entity:runCamPhysics(ID:number, Active:number, Velocity:vector) {
    if (!This) {
        return vec()
    }
    local LOCAL_BOUNDS = SEAT_BOUNDARIES[ID, array]
    
    if (!Active) {
        holoPos(ID, This:toWorld(vec(0,0,LOCAL_BOUNDS[3, number])))
        holoAng(ID, This:toWorld(ang()))
        return This:toWorld(vec(0,0,LOCAL_BOUNDS[3, number]))
    }

    #The prep
    local SID = "cam_"+ID
    if (!FUNC_DATA:exists(SID)) {
        #Variables: [1]=camera_pos, [2]=camera_velocity, [3]=seat_pos_idx
        FUNC_DATA[SID, array] = array(vec(), vec(), SEAT_POSITIONS[1, array][1, number])
    }
    local FrontBounds = LOCAL_BOUNDS[1, vector]
    local RearBounds  = LOCAL_BOUNDS[2, vector]
    local LOCAL_DATA  = FUNC_DATA[SID, array]
    local CamPos      = LOCAL_DATA[1, vector] #in <=> source units
    local CamVel      = LOCAL_DATA[2, vector] #m/s


    #The physics
    local Accel = vecSmooth( tickDiff("accel_"+ID, Velocity) * TickFreq, 3, "accel_"+ID, vec() )
    local AccelForce = BodyMass * Accel
    local DampingForce = CamVel * BodyMass * Damping
    local OutOfBoundsForce = vecDeadzone(CamPos, FrontBounds, RearBounds) * BodyMass * Damping * 0.5
    local BodyForce = vecDeadzone(CamPos, vec(-0.1), vec(0.1)) * C_in2m * BodyMass * BodyCompensation

    CamVel += (AccelForce - BodyForce - OutOfBoundsForce - DampingForce) / BodyMass * Tickrate
    CamPos += CamVel * C_m2in * Tickrate
    local SeatLocalCamPos = This:toLocal(CamPos + This:pos())
    local CamAng = ang(atan(SeatLocalCamPos:x() / LOCAL_BOUNDS[4, number]) * AccelViewRotation:x(), 0, -atan(SeatLocalCamPos:y() / LOCAL_BOUNDS[4, number]) * AccelViewRotation:y())


    #The results
    LOCAL_DATA[1, vector] = CamPos
    LOCAL_DATA[2, vector] = CamVel

    local WorldCamPos = This:pos() + CamPos + vec(0,0,LOCAL_BOUNDS[3, number]) + FUNC_DATA["cam_lerp_"+ID, array][1, vector]
    holoPos(ID, WorldCamPos)
    holoAng(ID, This:toWorld(CamAng))

    return WorldCamPos
}

#Handles stuff like forced exit from seats.
function number entity:handlePlayerInteractions(ID:number) {
    local SID = "cam_"+ID
    if (!This) {
        return 0
    }
    local LOCAL_DATA = FUNC_DATA[SID, array]
    local CurPos = LOCAL_DATA[3, number]
    local POS_DATA = SEAT_POSITIONS[ID, array]

    #EJ(&)Ection
    local Driver = This:driver()
    local BeingDriven = Driver:isValid() & Driver:isPlayer()
    if (BeingDriven & Driver:keyPressed(KeyEject)) {
        This:ejectPod()
    }

    #Moving the seat position
    if (BeingDriven & tickDiff("seat_move_up_"+SID, Driver:keyPressed(MovePositionUp)) & CurPos < (POS_DATA:count() - 1)) {
        CurPos++
    }
    if (BeingDriven & tickDiff("seat_move_down_"+SID, Driver:keyPressed(MovePositionDown)) & CurPos > 1) {
        CurPos--
    }
    interpolateCameraMovement(ID, POS_DATA[1 + CurPos, vector])

    LOCAL_DATA[3, number] = CurPos
    return BeingDriven
}

#Returns the velocity of the origin of the SEAT entity relative to the passed entity in world-space. The velocity is in m/s.
function vector entity:calcSeatWorldVelocity(SEAT:entity) {
    if (!SEAT | !SEAT:isValid() | !This | !This:isValid()) {
        return vec()
    }
    return vecSmooth( -This:velAtPoint(SEAT:pos()), VelocitySmoothing, "vel_seat_"+SEAT:id(), vec() ) * C_in2m
}

    C2Valid = C2_CTRLW:setUpCamCtrlWires(2)
    C3Valid = C3_CTRLW:setUpCamCtrlWires(3)

    SEAT1 = (C1Valid ? SEAT1 : _NO_ENTITY)
    SEAT1:createCameraParent(1)
    SEAT2 = (C2Valid ? SEAT2 : _NO_ENTITY)
    SEAT2:createCameraParent(2)
    SEAT3 = (C3Valid ? SEAT3 : _NO_ENTITY)
    SEAT3:createCameraParent(3)
}


#The device:
#-----------
function void main() {
    #Velocities
    local Seat1Vel = LOCO:calcSeatWorldVelocity(SEAT1)
    local Seat2Vel = LOCO:calcSeatWorldVelocity(SEAT2)
    local Seat3Vel = LOCO:calcSeatWorldVelocity(SEAT3)

    #Physics
    if (C1Valid) {
        C1_Active = SEAT1:handlePlayerInteractions(1)
        C1_Pos = SEAT1:runCamPhysics(1, C1_Active, Seat1Vel)
    }

    if (C2Valid) {
        C2_Active = SEAT2:handlePlayerInteractions(2)
        C2_Pos = SEAT2:runCamPhysics(2, C2_Active, Seat2Vel)
    }

    if (C3Valid) {
        C3_Active = SEAT3:handlePlayerInteractions(3)
        C3_Pos = SEAT3:runCamPhysics(3, C3_Active, Seat3Vel)
    }
}

if (first()|dupefinished()) {
    if(!timerExists("clk")){
        timer("clk", Tickrate, 0, function(){
            main()
        })
    }
}
